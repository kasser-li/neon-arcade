## 🎯 改进方案 (优先级 1)

### 1. 伤害飘字特效 ✨
**效果:** 攻击时显示飘动的伤害数字

**实现代码:**
```javascript
// 添加伤害数字
function addDamageNumber(x, y, text, isPlayer, isCrit = false) {
    gameState.damageNumbers.push({
        x, y, text, isPlayer, isCrit,
        life: 60,  // 存活 60 帧
        size: isCrit ? 36 : 24
    });
}

// 绘制伤害数字
function drawDamageNumbers() {
    for (let i = gameState.damageNumbers.length - 1; i >= 0; i--) {
        const dn = gameState.damageNumbers[i];
        ctx.save();
        ctx.translate(dn.x, dn.y);
        ctx.globalAlpha = dn.life / 60;
        ctx.fillStyle = dn.isCrit ? '#ff0' : (dn.isPlayer ? '#fff' : '#ff6666');
        ctx.font = `bold ${dn.size || 24}px sans-serif`;
        ctx.fillText(dn.text, 0, 0);
        ctx.restore();
        
        dn.y -= 2;  // 向上飘动
        dn.life--;
        if (dn.life <= 0) {
            gameState.damageNumbers.splice(i, 1);
        }
    }
}
```

**使用示例:**
```javascript
// 攻击时调用
addDamageNumber(canvas.width / 2, 150, `-${damage}`, true);
```
